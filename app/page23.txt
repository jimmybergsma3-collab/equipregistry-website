export const dynamic = "force-dynamic";

type Props = {
  searchParams: Promise<{
    serial?: string;
  }>;
};

function getStatus(serial: string) {
  const s = serial.trim().toUpperCase();

  if (s === "ER-REG-001") {
    return {
      label: "Registered & Verified",
      color: "green",
      message:
        "This machine is registered in EquipRegistry and its legal origin has been verified.",
      why:
        "This equipment has a verified legal origin and an active registry passport. It can be insured, sold or rented with reduced legal and insurance risk.",
    };
  }

  if (s === "ER-HIS-404") {
    return {
      label: "History Unknown",
      color: "orange",
      message:
        "This machine is registered, but its full ownership history could not be verified.",
      why:
        "Incomplete ownership history increases fraud, insurance and compliance risk. Verification reduces uncertainty for insurers and buyers.",
    };
  }

  if (s === "ER-STOL-777") {
  return {
    label: "Stolen Equipment – Red Flag",
    color: "red",
    message:
      "This machine has been officially reported stolen and is actively blacklisted in EquipRegistry.",
    metadata: [
      { label: "Status", value: "Blacklisted" },
      { label: "Risk level", value: "High" },
      { label: "Reported by", value: "Insurance partner" },
      { label: "Jurisdiction", value: "EU / Cross-border alert" },
      { label: "Report date", value: "2025-03-12" },
    ],
    warning:
      "Do NOT purchase, insure, rent or transport this machine.",
    why:
      "Equipment listed as stolen represents a high legal, financial and insurance risk. Any transaction involving this asset may lead to claim denial, liability exposure, cross-border seizure or criminal investigation.",
    actions: [
      { label: "Report sighting", style: "primary" },
      { label: "Contact authorities", style: "danger" },
      { label: "Verify case ID", style: "secondary" },
    ],
  };
}


  return {
    label: "Not Registered",
    color: "gray",
    message:
      "This serial number is not registered in EquipRegistry.",
    why:
      "Unregistered equipment lacks a verified ownership record, increasing legal and insurance uncertainty.",
  };
}
